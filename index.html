    Services

    Encapsulate the responsibilities related to each other and business logic and achieve single goal

    A service has only and only one instance throughout the application
    A service is called as :
    myApp = angular.module("myApp",[]);

    myApp.service("addSvc",function(){
          this.add=function(firstNum,secondNum){
                console.log('Inside add service')
                return parseInt(firstNum)+parseInt(secondNum);
          }
    })

    myApp.controller('AppCtrl',function($scope,addSvc){
          $scope.calculate=function(firstNum,secondNum){
                console.log('Calculating via service ...')
                this.result=addSvc.add(firstNum,secondNum);
          }
    })
